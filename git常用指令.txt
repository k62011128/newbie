//下载依赖包(所有从githua下载下来的项目都要在终端执行此命令)
npm install

//查看本地git状态
git status

//把更新的代码添加到暂存区
git add *

//将暂存区的更新提交到本地仓库
git commit -m "更新说明"

//查看所有分支
git branch -a

//创建本地分支x
git branch x

//删除本地分支x
git branch -d x

//切换到本地分支x
git checkout x

//本地分支改名
git branch -m old_branch new_branch

//创建远程分支x并将本地分支x推送到远程分支x(冒号前是本地分支，冒号后是远程分支,远程分支没有会自动创建)
git push origin dev:dev

//删除远程分支(即推送空分支到远程分支)
git push origin :dev

//下面介绍如何拉取项目，先新建文件夹
git clone https://github.com/id/仓库名.git
cd 仓库名

//下面介绍如何更新并上传项目
//如果是第一次上传项目要加以下指令
git init
git remote add origin https://github.com/id/仓库名.git     //自己仓库地址
//不是第一次直接用下面的指令
git add *
git commit -m "更新说明"
git pull origin main//必须要先同步远程分支，不同步有可能传不上去
git push origin main//如果还是传不上去报同步错误可以用git push -f origin main强制覆盖远程代码，缺点是以后工作里不小心用了就等同于删库跑路
//如果上传失败报错了，自己百度，每个人的问题不一定一样

//解除ssl验证,我的上传报错是用下面指令解决问题的
git config --global http.sslVerify "false"
git config --global http.proxy
git config --global --unset http.proxy
